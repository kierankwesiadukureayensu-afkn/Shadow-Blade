// ===== IMPORTS ===== const http = require("http"); const WebSocket = require("ws"); // ===== HTTP SERVER (SERVES GAME) ===== const server = http.createServer((req, res) => { res.writeHead(200, {"Content-Type":"text/html"}); res.end(`
`); }); // ===== WEBSOCKET SERVER ===== const wss = new WebSocket.Server({ server }); // ===== WORLD STATE ===== let players = {}; let fruits = []; let bosses = [{id:1,x:500,y:300,hp:500}]; let projectiles = []; // ===== FRUIT SPAWNER ===== setInterval(()=>{ fruits.push({ id:Math.random().toString(36).substr(2,6), x:Math.random()*800, y:Math.random()*600 }); },15000); // ===== GAME LOOP ===== setInterval(()=>{ // move projectiles projectiles.forEach(p=>{ p.x += p.vx; p.y += p.vy; // hit players Object.entries(players).forEach(([id,pl])=>{ if(id!==p.owner && Math.hypot(p.x-pl.x,p.y-pl.y)<15){ pl.hp -= 10; } }); // hit bosses bosses.forEach(b=>{ if(Math.hypot(p.x-b.x,p.y-b.y)<30){ b.hp -= 15; } }); }); // regen energy Object.values(players).forEach(p=>{ p.energy = Math.min(100, p.energy + 0.5); }); broadcast(); },50); // ===== CONNECTION ===== wss.on("connection", ws=>{ let id = Math.random().toString(36).substr(2,9); players[id] = { x:100, y:100, hp:100, energy:100, fruit:null, mastery:1 }; ws.send(JSON.stringify({type:"init",id})); ws.on("message", msg=>{ let d = JSON.parse(msg); let p = players[id]; if(!p) return; // movement if(d.type==="move"){ p.x = d.x; p.y = d.y; } // fruit pickup if(d.type==="pickup"){ let i = fruits.findIndex(f=>f.id===d.id); if(i!==-1){ p.fruit = "Fire"; fruits.splice(i,1); } } // ability if(d.type==="ability"){ if(p.energy < 10) return; p.energy -= 10; p.mastery += 0.2; // shoot projectile projectiles.push({ x:p.x, y:p.y, vx:(Math.random()-0.5)*6, vy:(Math.random()-0.5)*6, owner:id }); } }); // disconnect ws.on("close",()=>delete players[id]); }); // ===== BROADCAST ===== function broadcast(){ let data = JSON.stringify({players,fruits,bosses,projectiles}); wss.clients.forEach(c=>c.send(data)); } // ===== START SERVER ===== server.listen(3000,()=>{ console.log("🌍 MMO running at http://localhost:3000"); });
